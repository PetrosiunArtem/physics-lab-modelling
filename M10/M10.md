$$ E = -B \sum{s_i} = -B (N_+ - N_-) $$

(считаем $ B > 0 $)

Выбрали со спином +1 $ \implies \Delta E = (-B (N_+ - 1 - N_- - 1) + B (N_+ - N_-)) = 2B > 0 $

Выбрали со спином -1 $ \implies \Delta E = (-B (N_+ + 1 - N_- + 1) + B (N_+ - N_-)) = -2B < 0 $

$$ \begin{cases}
\dot N _+ = \frac {N_-} N - e^{\frac {-2B} T} \frac {N_+} N \\
\dot N _- = - \frac {N_-} N + e^{\frac {-2B} T} \frac {N_+} N \\
\end{cases} $$

$$ \dot N _+ = 1 - (e^{\frac {-2B} T} + 1) \frac {N_+} N $$

$$ \frac {N_+} N = \frac 1 {e^{\frac{-2B} T} + 1} $$
$$ \frac {N_+} N - \frac {N_-} N = 2 \frac {N_+} N - 1 = \frac 2 {e^{\frac {-2B} T} + 1} - 1 = \frac {1 - e^{\frac {-2B} T}} {1 + e^{\frac {-2B} T}} = -\tanh{\frac {-B} T} = \tanh{\frac B T} $$

$$ S \rightarrow max $$
$$ - N_+ \ln{\frac {N_+} N} - N_- \ln{\frac {N_-} N} \rightarrow \max $$
$$ - N_+ \ln{\frac {N_+} N} - (N - N_+) \ln{\frac {N - N_+} N} \rightarrow \max $$
$$ -\ln \frac {N_+} N - 1 + \ln \frac {N_-} N + 1 = \ln \frac {N_-} {N_+} = 0 $$
$$ N_- = N_+ = N / 2 \iff M = N $$
Достигается при $ T \rightarrow \infty $
